@app.route('/predict_bulk', methods=['POST'])
def predict_bulk():
    # ... (file check wala part same rahega) ...
    try:
        df = pd.read_csv(file)

        # Galti 1: Purane Index column ko hatayein
        if 'Unnamed: 0' in df.columns:
            df.drop(columns=['Unnamed: 0'], inplace=True)

        # Galti 2: Case-insensitive mapping (Male/male/M sab handle hoga)
        df['Sex'] = df['Sex'].astype(str).str.strip().str.capitalize()
        df['ChestPainType'] = df['ChestPainType'].astype(str).str.strip().str.upper()
        df['RestingECG'] = df['RestingECG'].astype(str).str.strip().str.capitalize()
        df['ExerciseAngina'] = df['ExerciseAngina'].astype(str).str.strip().str.upper()
        df['ST_Slope'] = df['ST_Slope'].astype(str).str.strip().str.capitalize()

        # Mapping dictionaries (Professional Handle)
        sex_map = {'Male': 1, 'M': 1, '1': 1, 'Female': 0, 'F': 0, '0': 0}
        cp_map = {'TA': 0, 'ATA': 1, 'NAP': 2, 'ASY': 3}
        ecg_map = {'Normal': 0, 'ST': 1, 'LVH': 2}
        angina_map = {'Y': 1, 'Yes': 1, 'N': 0, 'No': 0}
        slope_map = {'Up': 0, 'Flat': 1, 'Down': 2}

        # Apply Mapping
        df['sex_num'] = df['Sex'].map(sex_map)
        df['cp_num'] = df['ChestPainType'].map(cp_map)
        df['ecg_num'] = df['RestingECG'].map(ecg_map)
        df['angina_num'] = df['ExerciseAngina'].map(angina_map)
        df['slope_num'] = df['ST_Slope'].map(slope_map)

        # Model ke liye 11 features ka sahi order
        input_cols = {
            'Age': 'age', 'sex_num': 'sex', 'cp_num': 'cp', 
            'RestingBP': 'trestbps', 'Cholesterol': 'chol', 'FastingBS': 'fbs',
            'ecg_num': 'restecg', 'MaxHR': 'thalach', 'angina_num': 'exang',
            'Oldpeak': 'oldpeak', 'slope_num': 'slope'
        }
        
        # Prediction ke liye temporary dataframe
        X = df[list(input_cols.keys())].rename(columns=input_cols)
        
        # Predict
        probabilities = model.predict_proba(X)[:, 1] * 100
        df['Risk_Score'] = probabilities.round(2)
        df['Result'] = ['High Risk 🚨' if p >= 50 else 'Low Risk ✅' for p in probabilities]

        # Galti 3: Sirf kaam ke columns display karein (Extra 'ca', 'thal' hata dein)
        display_df = df[['Age', 'Sex', 'ChestPainType', 'RestingBP', 'Cholesterol', 'Risk_Score', 'Result']]

        result_path = "static/bulk_results.csv"
        df.to_csv(result_path, index=False)

        return render_template('bulk_result.html', 
                               tables=[display_df.to_html(classes='data premium-table', index=False)], 
                               file_path=result_path)

    except Exception as e:
        return f"Mapping Error: {str(e)}. Check if CSV values match (e.g., Male/Female, TA/ATA/NAP/ASY)"
